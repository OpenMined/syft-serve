The Complete API

effortless creation and management of fastapi servers

Creating Servers

import syft_serve as ss

# Simple server
server = ss.create(
    name="my_api",
    endpoints={"/hello": hello_func}
)

Server Management

# View all servers
ss.servers

# Access by name
server = ss.servers["my_api"]
print(server.url)  # http://localhost:8000

Viewing Logs

# Easy log access
server.stdout.tail(20)    # Last 20 lines
server.stderr.head(10)    # First 10 error lines
server.uptime             # "2m 15s"

Clean Shutdown

# Terminate one server
server.terminate()

# Terminate all servers
ss.terminate_all()
